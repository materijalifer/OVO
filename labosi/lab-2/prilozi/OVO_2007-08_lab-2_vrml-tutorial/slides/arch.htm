Transforming shapes
arch.wrl
 

	

#VRML V2.0 utf8
#
#  Simple arch
#  arch.wrl
#        by David R. Nadeau
#        published in "The VRML 2.0 Sourcebook" by Ames, Nadeau, and Moreland
#
#  Illustrate the use of translation and rotation to position and
#  orient shapes.
#

# Left and right columns
Transform {
    translation -2.0 -1.0 0.0
    children [
        DEF Column Shape {
            appearance DEF White Appearance {
                material Material { }
            }
            geometry Cylinder {
                radius 0.3
                height 6.0
                top FALSE
            }
        }
    ]
}
Transform {
    translation 2.0 -1.0 0.0
    children [ USE Column ]
}

# Cross-piece
Transform {
    translation 0.0 2.05 0.0
    children [
        Shape {
            appearance USE White
            geometry Box { size 4.6 0.4 0.6 }
        }
    ]
}

# Roof pieces
Transform {
    translation -1.15 3.12 0.0
    rotation 0.0 0.0 1.0  0.524
    children [
        DEF Roof Shape {
            appearance USE White
            geometry Box { size 2.86 0.4 0.6 }
        }
    ]
}
Transform {
    translation 1.15 3.12 0.0
    rotation 0.0 0.0 1.0  -0.524
    children [ USE Roof ]
}
